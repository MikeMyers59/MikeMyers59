    d3.csv("watch.csv", function(watch) {
      


      var sampleSVG = d3.select(".example_div")
        .append("svg:svg")
        .attr("class", "sample")
        .attr("width", 300)
        .attr("height", 300);

      sampleSVG
        .selectAll("circle")
        .data(watch)
        .enter()
        .append("svg:circle")
        .attr("stroke", "black")
        .attr("fill", "aliceblue")
        .attr("r", function(d) {
          console.log(d)
          return +d.size;
        })
        .attr("cx", function(d,i){ return 52 + (40*i)})
        .attr("cy", 52)

    $('svg circle').tipsy({ 
        gravity: 'w', 
        html: true, 
        title: function() {
          var d = this.__data__;
          return 'Hi there! My color is <span style="color:' + d.circle + '">' + d.size + '</span>'; 
        }
      });


    });
